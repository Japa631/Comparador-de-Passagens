const form=document.getElementById('searchForm');

const results=document.getElementById('results');

const btnGenerate=document.getElementById('generate');

const btnOpenAll=document.getElementById('openAll');

const btnClear=document.getElementById('clear');



form.addEventListener('submit',function(ev){

ev.preventDefault();

const origin=document.getElementById('origin').value;

const destination=document.getElementById('destination').value;

const departDate=document.getElementById('departDate').value;

const returnDate=document.getElementById('returnDate').value;

const adults=document.getElementById('adults').value;

const cabin=document.getElementById('cabin').value;

if(!origin||!destination||!departDate){alert('Preencha Origem, Destino e Data de Ida.');return;}

const links=buildLinks({origin,destination,departDate,returnDate,adults,cabin});

results.innerHTML='';

links.forEach(l=>{

const div=document.createElement('div');div.className='provider';

const left=document.createElement('div');left.innerHTML=`<b>${l.name}</b><div class="small">${l.url}</div>`;

const right=document.createElement('div');

const a=document.createElement('a');a.href=l.url;a.target='_blank';a.rel='noopener noreferrer';a.textContent='Abrir';a.style.marginRight='8px';

const copy=document.createElement('button');copy.type='button';copy.textContent='Copiar URL';copy.className='secondary';

copy.addEventListener('click',()=>{navigator.clipboard.writeText(l.url).then(()=>alert('URL copiada!'))});

right.appendChild(a);right.appendChild(copy);

div.appendChild(left);div.appendChild(right);

results.appendChild(div);

});

});



btnOpenAll.addEventListener('click',()=>{

const origin=document.getElementById('origin').value;

const destination=document.getElementById('destination').value;

const departDate=document.getElementById('departDate').value;

const adults=document.getElementById('adults').value;

const cabin=document.getElementById('cabin').value;

const links=buildLinks({origin,destination,departDate,returnDate,adults,cabin});

links.forEach(l=>{window.open(l.url,'_blank');});

});



btnClear.addEventListener('click',()=>{
